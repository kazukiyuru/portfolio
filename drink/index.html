自分の酒飲み

GAS × Google スライド 連携アプリ

概要

自分の酒飲みをランキング化しました。
スプレッドシートで統計を取り、ピボットテーブルで本数をランキング化。
→ グラフにして GAS を用いて Google スライドに自動貼り付け。

使用ツール

Google スプレッドシート ピボットテーブル Google Apps Script Google スライド

リンク

📊 スプレッドシート 📽️ Google スライド

GAS コード

function createSlide() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName("ピボット テーブル  1");
  const rows = sheet.getRange(1,1,6,2).getValues();

  const most   = rows[2][0];
  const second = rows[3][0];
  const out    = rows[1][0];
  const high   = rows[4][0];

  const comment = `最も飲んだのは${most}。2番目に${second}。${out},${high}は外呑み。`;

  const chart = sheet.getCharts()[0];

  // 新規スライドを作成
  const presentation = SlidesApp.create("自分の酒飲みランキング");
  const slide = presentation.getSlides()[0];

  // タイトルを挿入
  const titleBox = slide.insertTextBox("自分の酒飲みランキング");
  titleBox.setLeft(50).setTop(20).setWidth(600).setHeight(50);
  titleBox.getText().getTextStyle().setFontSize(24).setBold(true);

  // コメントを挿入
  const commentBox = slide.insertTextBox(comment);
  commentBox.setLeft(50).setTop(80).setWidth(620).setHeight(40);
  commentBox.getText().getTextStyle().setFontSize(14);

  // グラフを画像として挿入
  const image = slide.insertSheetsChartAsImage(chart);
  image.scaleHeight(0.5)
       .scaleWidth(0.5)
       .alignOnPage(SlidesApp.AlignmentPosition.HORIZONTAL_CENTER)
       .setTop(130);

  // 作成したスライドのURLをログに出力
  Logger.log(presentation.getUrl());
}

createSlide();
